# Skill: HMAC-SHA256 Signed Evidence Export

> **Roadmap refs:** H5, §21.1-H  
> **Complexity:** Medium  
> **Files affected:** `extension/sidepanel.js`, new `extension/lib/evidence-pack.js`

---

## Problem

The extension exports a raw JSON dump (`lss_results`). The roadmap defines a
signed evidence pack format (`nxlv-evidence-pack/v1`) authenticated with
HMAC-SHA256 so recipients can verify the report was generated by this tool
and wasn't tampered with.

---

## Implementation

### 1. `extension/lib/evidence-pack.js`

```js
// Pure, no chrome.* calls, no fetch

const PACK_VERSION = 'nxlv-evidence-pack/v1';

/**
 * Generate an HMAC-SHA256 key from a passphrase using PBKDF2.
 * If no passphrase given, uses a deterministic device key from a random uuid
 * stored in chrome.storage.
 */
export async function deriveSigningKey(passphrase) {
  const enc = new TextEncoder();
  const base = await crypto.subtle.importKey(
    'raw', enc.encode(passphrase), { name: 'PBKDF2' }, false, ['deriveKey']
  );
  return crypto.subtle.deriveKey(
    { name: 'PBKDF2', salt: enc.encode('lpa-evidence-v1'), iterations: 100000, hash: 'SHA-256' },
    base,
    { name: 'HMAC', hash: 'SHA-256' },
    true,
    ['sign', 'verify']
  );
}

/**
 * Build and sign an evidence pack.
 * Returns { pack: object, signature: hex string, publicKey: null }
 */
export async function buildEvidencePack(results, summary, signingKey) {
  const pack = {
    version: PACK_VERSION,
    generated_at: new Date().toISOString(),
    tool: 'lovable-portfolio-audit',
    tool_version: '1.0.0',
    summary: {
      total_projects: summary.totalProjects,
      scanned_projects: summary.scannedProjects,
      critical: summary.criticalCount,
      high: summary.highCount,
      medium: summary.mediumCount,
      low: summary.lowCount,
      clean: summary.cleanCount,
      scan_start: summary.scanStartTime,
      scan_end: summary.scanEndTime,
    },
    findings: results.map(r => ({
      project_id: r.projectId,
      project_name: r.projectName,
      severity: r.severity,
      risk_score: r.riskScore,
      probe_results: r.probeResults || [],
      findings: r.findings.map(f => ({
        id: f.id,
        rule_id: f.ruleId,
        severity: f.severity,
        title: f.title,
        vector: f.vector,
        source: f.source,
        evidence: f.evidence,          // masked only
        secret_hash: f.secret_hash,    // sha256[:16], never raw
        line: f.line,
        recommendation: f.recommendation,
      })),
      supabase_detected: r.supabaseDetected,
      files_scanned: r.filesScanned,
      chat_messages_scanned: r.chatMessagesScanned,
    })),
  };

  const payload = JSON.stringify(pack, null, 2);
  const sig = await crypto.subtle.sign(
    'HMAC',
    signingKey,
    new TextEncoder().encode(payload)
  );
  const sigHex = Array.from(new Uint8Array(sig))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');

  return { pack, payload, signature: sigHex };
}

/**
 * Verify a previously exported pack.
 * Returns true/false.
 */
export async function verifyEvidencePack(payload, sigHex, signingKey) {
  const sigBytes = new Uint8Array(sigHex.match(/.{2}/g).map(h => parseInt(h, 16)));
  return crypto.subtle.verify('HMAC', signingKey, sigBytes, new TextEncoder().encode(payload));
}
```

### 2. Device signing key (no passphrase fallback)

In `sidepanel.js`, when user hasn't set a passphrase, derive from a random
device key stored in `chrome.storage.local`:

```js
async function getOrCreateDeviceKey() {
  const stored = await chrome.storage.local.get('lpa_device_key');
  if (stored.lpa_device_key) return stored.lpa_device_key;
  const key = crypto.randomUUID() + '-' + crypto.randomUUID();
  await chrome.storage.local.set({ lpa_device_key: key });
  return key;
}
```

### 3. Export UI in `sidepanel.html`

```html
<div id="export-section">
  <h3>Export Report</h3>
  <button id="export-json-btn">Download JSON (unsigned)</button>
  <button id="export-signed-btn">Download Signed Evidence Pack</button>
  <details>
    <summary>Custom signing passphrase (optional)</summary>
    <input id="sign-passphrase" type="password"
      placeholder="Leave blank to use device key" />
    <small>The passphrase is used with PBKDF2 to derive an HMAC-SHA256 key.
      Keep it safe — you'll need it to verify the pack later.</small>
  </details>
</div>
```

```js
document.getElementById('export-signed-btn').addEventListener('click', async () => {
  const { lss_results, lss_summary } = await chrome.storage.local.get(['lss_results', 'lss_summary']);
  const passphrase = document.getElementById('sign-passphrase').value.trim()
    || await getOrCreateDeviceKey();
  const key = await deriveSigningKey(passphrase);
  const { payload, signature } = await buildEvidencePack(lss_results || [], lss_summary || {}, key);

  const output = JSON.stringify({ payload: JSON.parse(payload), hmac_sha256: signature }, null, 2);
  downloadFile(output, `lpa-evidence-${Date.now()}.json`, 'application/json');
});
```

## Acceptance Criteria

- [ ] Signed export contains `hmac_sha256` field (64-char hex)
- [ ] Evidence pack never contains raw secret values (only `evidence` = masked, `secret_hash` = sha256[:16])
- [ ] `verifyEvidencePack()` returns `true` for a freshly exported pack
- [ ] `verifyEvidencePack()` returns `false` if payload is tampered
- [ ] Device key stored in `chrome.storage.local` survives extension restart
- [ ] Passphrase input is type=password (not shown in plain text)
